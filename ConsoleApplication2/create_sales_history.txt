
SET hbase.zookeeper.quorum=zookeepernode0,zookeepernode1,zookeepernode2;  
CREATE EXTERNAL TABLE SalesHistory(
		SaleID string,
		Product string,
		SaleDate string,
		StatusID int,
		SalePrice double
		)
STORED BY 'org.apache.hadoop.hive.hbase.HBaseStorageHandler'
WITH SERDEPROPERTIES ('hbase.columns.mapping' = ':key, cf:Product, cf:SaleDate, cf:StatusID, cf:SalePrice' )
TBLPROPERTIES ('hbase.table.name' = 'SalesHistory');
